void leaves(Node* root,int &c)
{
    if(root==NULL)
    return ;
    leaves(root->left,c);
    leaves(root->right,c);
    if(root->left==NULL && root->right==NULL)
    c+=1;
}
int countNodes(Node* root)
{
    if(root==NULL)
    return 0;
    return 1+countNodes(root->left)+countNodes(root->right);
}
int countNonLeafNodes(Node* root)
{
    // Code here
    int c=0;
    leaves(root,c);
    int total=countNodes(root);
    return total-c;
    
}
